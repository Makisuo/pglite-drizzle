<Step>

### Installation

<Tabs groupId='package-manager' items={['npm', 'pnpm', 'yarn', 'bun']} persist>
  ```bash tab="npm"
  npx jsr add @makisuo/pglite-drizzle
  ```

  ```bash tab="pnpm"
  pnpm dx jsr add @makisuo/pglite-drizzle
  ```

  ```bash tab="yarn"
  yarn dlx jsr add @makisuo/pglite-drizzle
  ```

  ```bash tab="bun"
  bunx jsr add @makisuo/pglite-drizzle
  ```
</Tabs>

</Step>
<Step>

### Initialize Client

```ts title="drizzle-client.ts"
import { createDrizzle } from "@makisuo/pglite-drizzle/react"

// Your drizzle schema
import { schema } from "./schema.ts"

export const { useDrizzleLive, useDrizzleLiveIncremental } = createDrizzle({ schema })
```

</Step>
 
<Step>
 
### Usage

```tsx title="page.tsx"
import { useDrizzleLive } from "./drizzle-client.ts"

const { data } = useDrizzleLive((db) =>
        db.query.accounts.findMany({
            with: {
                institution: true,
            },
        }),
)
```
</Step>
