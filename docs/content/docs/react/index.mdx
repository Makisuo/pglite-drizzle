## Setup

```ts title="drizzle-client.ts"
import { createDrizzle } from "@makisuo/pglite-drizzle/react"

// Your drizzle schema
import { schema } from "./schema.ts"

export const { useDrizzleLive, useDrizzleLiveIncremental } = createDrizzle({ schema })
```

## Usage
```tsx title="page.tsx"
import { useDrizzleLive } from "./drizzle-client.ts"

const { data } = useDrizzleLive((db) =>
        db.query.accounts.findMany({
            with: {
                institution: true,
            },
        }),
)
```
